/******************************************************************************
 * Formula Name : HRX_VARIABLE_DAY_DAILY_OVERTIME_CALCULATOR
 * Formula Type : Calculation Utility
 * Description  : Calculates daily overtime using a variable 24-hour workday.
 *                A workday is a rolling 24-hour window beginning from the
 *                first time entry start time after the prior workday ends.
 *                Time is split across workday boundaries and allocated into
 *                regular and overtime measures based on the defined limit.
 * Change History
 * --------------
 *  Who             Ver    Date          Description
 * ---------------  ----   -----------   --------------------------------
 * Payroll Admin    1.0    2026/04/10    Created.
 ******************************************************************************/

DEFAULT FOR HWM_CTXARY_RECORD_POSITIONS IS EMPTY_TEXT_NUMBER
DEFAULT FOR HWM_CTXARY_HWM_MEASURE_DAY IS EMPTY_NUMBER_NUMBER
DEFAULT FOR MEASURE IS EMPTY_NUMBER_NUMBER
DEFAULT FOR STARTTIME IS EMPTY_DATE_NUMBER
DEFAULT FOR STOPTIME IS EMPTY_DATE_NUMBER
DEFAULT FOR TIMERECORDTYPE IS EMPTY_TEXT_NUMBER

INPUTS ARE
  HWM_CTXARY_RECORD_POSITIONS,
  HWM_CTXARY_HWM_MEASURE_DAY,
  MEASURE,
  STARTTIME,
  STOPTIME,
  TIMERECORDTYPE

LOCAL FF_NAME (TEXT)
LOCAL NULLDATE (DATE)
LOCAL NULLDATETIME (DATE)
LOCAL NULLTEXT (TEXT)
LOCAL REC_POSITION_DETAIL (TEXT)
LOCAL TIMERECORDTYPE_RANGE (TEXT)
LOCAL CTIME_MULTI_HOURS (NUMBER)
LOCAL P_DAILY_LIMIT_HRS (NUMBER)
LOCAL TCRECCOUNT (NUMBER)
LOCAL CURDAYSTART (DATE)
LOCAL CURDAYEND (DATE)
LOCAL CURDAYTOTAL (NUMBER)
LOCAL WEEKSTART (DATE)
LOCAL WEEKDAYIDX (NUMBER)
LOCAL WORKDAYSEQ (NUMBER)
LOCAL NIDX (NUMBER)
LOCAL RECPOS (TEXT)
LOCAL TCMEASURE (NUMBER)
LOCAL TCSTART (DATE)
LOCAL TCSTOP (DATE)
LOCAL TCTYPE (TEXT)
LOCAL OUTREG (NUMBER)
LOCAL OUTOT (NUMBER)
LOCAL SEGSTART (DATE)
LOCAL SEGSTOP (DATE)
LOCAL SEGHRSFALLBACK (NUMBER)
LOCAL REGCAP (NUMBER)
LOCAL REGPART (NUMBER)
LOCAL OTPART (NUMBER)
LOCAL SPLITGUARD (NUMBER)
LOCAL SEGDAYSTOP (DATE)
LOCAL SEGHRS (NUMBER)
LOCAL MEASURE_UNDER (NUMBER_NUMBER)
LOCAL MEASURE_OVER (NUMBER_NUMBER)

FF_NAME = 'HRX_VARIABLE_DAY_DAILY_OVERTIME_CALCULATOR'
PAY_INTERNAL_LOG_WRITE('>>> Enter - ' || FF_NAME)

NULLDATE = '01-JAN-1900'(DATE)
NULLDATETIME = '1900/01/01 00:00:00'(DATE)
NULLTEXT = '**FF_NULL**'

REC_POSITION_DETAIL = 'DETAIL'
TIMERECORDTYPE_RANGE = 'RANGE'
CTIME_MULTI_HOURS = 24

P_DAILY_LIMIT_HRS = 8

MEASURE_UNDER = EMPTY_NUMBER_NUMBER
MEASURE_OVER = EMPTY_NUMBER_NUMBER

TCRECCOUNT = HWM_CTXARY_RECORD_POSITIONS.COUNT

CURDAYSTART = NULLDATE
CURDAYEND = NULLDATE
CURDAYTOTAL = 0

WEEKSTART = NULLDATE
WEEKDAYIDX = 0
WORKDAYSEQ = 0

NIDX = 0
WHILE (NIDX < TCRECCOUNT) LOOP
(
  NIDX = NIDX + 1

  IF (NIDX > 1000) THEN
  (
    PAY_INTERNAL_LOG_WRITE('Loop guard exceeded in ' || FF_NAME)
    RETURN MEASURE_UNDER, MEASURE_OVER
  )

  RECPOS = NULLTEXT
  TCMEASURE = 0
  TCSTART = NULLDATE
  TCSTOP = NULLDATE
  TCTYPE = NULLTEXT

  IF (HWM_CTXARY_RECORD_POSITIONS.EXISTS(NIDX)) THEN
    RECPOS = HWM_CTXARY_RECORD_POSITIONS[NIDX]
  IF (MEASURE.EXISTS(NIDX)) THEN
    TCMEASURE = MEASURE[NIDX]
  IF (STARTTIME.EXISTS(NIDX)) THEN
    TCSTART = STARTTIME[NIDX]
  IF (STOPTIME.EXISTS(NIDX)) THEN
    TCSTOP = STOPTIME[NIDX]
  IF (TIMERECORDTYPE.EXISTS(NIDX)) THEN
    TCTYPE = TIMERECORDTYPE[NIDX]

  IF (RECPOS = REC_POSITION_DETAIL AND TCMEASURE > 0) THEN
  (
    OUTREG = 0
    OUTOT = 0

    SEGSTART = TCSTART
    SEGSTOP = TCSTOP
    SEGHRSFALLBACK = TCMEASURE

    IF (TCTYPE <> TIMERECORDTYPE_RANGE OR SEGSTART = NULLDATE OR SEGSTOP = NULLDATE OR SEGSTOP <= SEGSTART) THEN
    (
      SEGSTART = NULLDATE
      SEGSTOP = NULLDATE

      IF (CURDAYSTART = NULLDATE) THEN
      (
        CURDAYSTART = NULLDATE
        CURDAYEND = NULLDATE
        CURDAYTOTAL = 0
      )

      REGCAP = P_DAILY_LIMIT_HRS - CURDAYTOTAL
      IF (REGCAP < 0) THEN
        REGCAP = 0

      REGPART = LEAST(SEGHRSFALLBACK, REGCAP)
      OTPART = SEGHRSFALLBACK - REGPART

      CURDAYTOTAL = CURDAYTOTAL + SEGHRSFALLBACK
      OUTREG = OUTREG + REGPART
      OUTOT = OUTOT + OTPART

      PAY_INTERNAL_LOG_WRITE('Row ' || TO_CHAR(NIDX) ||
        '; noRange; hrs=' || TO_CHAR(SEGHRSFALLBACK) ||
        '; dayTotal=' || TO_CHAR(CURDAYTOTAL) ||
        '; reg=' || TO_CHAR(OUTREG) ||
        '; ot=' || TO_CHAR(OUTOT))
    )
    ELSE
    (
      SPLITGUARD = 0
      WHILE (SEGSTART < SEGSTOP) LOOP
      (
        SPLITGUARD = SPLITGUARD + 1

        IF (SPLITGUARD > 50) THEN
        (
          PAY_INTERNAL_LOG_WRITE('Split loop guard exceeded in ' || FF_NAME || '; row=' || TO_CHAR(NIDX))
          RETURN MEASURE_UNDER, MEASURE_OVER
        )

        IF (CURDAYSTART = NULLDATE OR SEGSTART >= CURDAYEND) THEN
        (
          CURDAYSTART = SEGSTART
          CURDAYEND = ADD_DAYS(CURDAYSTART, 1)
          CURDAYTOTAL = 0

          WORKDAYSEQ = WORKDAYSEQ + 1
          WEEKDAYIDX = WEEKDAYIDX + 1

          IF (WEEKSTART = NULLDATE OR WEEKDAYIDX > 7) THEN
          (
            WEEKSTART = CURDAYSTART
            WEEKDAYIDX = 1
          )

          PAY_INTERNAL_LOG_WRITE('New workday #' || TO_CHAR(WORKDAYSEQ) ||
            '; weekDay=' || TO_CHAR(WEEKDAYIDX) ||
            '; dayStart=' || TO_CHAR(CURDAYSTART) ||
            '; dayEnd=' || TO_CHAR(CURDAYEND))
        )

        SEGDAYSTOP = LEAST(SEGSTOP, CURDAYEND)
        SEGHRS = ROUND((DAYS_BETWEEN(SEGDAYSTOP, SEGSTART) * CTIME_MULTI_HOURS), 2)

        REGCAP = P_DAILY_LIMIT_HRS - CURDAYTOTAL
        IF (REGCAP < 0) THEN
          REGCAP = 0

        REGPART = LEAST(SEGHRS, REGCAP)
        OTPART = SEGHRS - REGPART

        CURDAYTOTAL = CURDAYTOTAL + SEGHRS
        OUTREG = OUTREG + REGPART
        OUTOT = OUTOT + OTPART

        PAY_INTERNAL_LOG_WRITE('Split' ||
          '; segStart=' || TO_CHAR(SEGSTART) ||
          '; segStop=' || TO_CHAR(SEGDAYSTOP) ||
          '; segHrs=' || TO_CHAR(SEGHRS) ||
          '; dayTotal=' || TO_CHAR(CURDAYTOTAL) ||
          '; regPart=' || TO_CHAR(REGPART) ||
          '; otPart=' || TO_CHAR(OTPART))

        SEGSTART = SEGDAYSTOP
      )
    )

    IF (OUTREG > 0) THEN
      MEASURE_UNDER[NIDX] = OUTREG
    IF (OUTOT > 0) THEN
      MEASURE_OVER[NIDX] = OUTOT
  )
)

PAY_INTERNAL_LOG_WRITE('<<< Exit - ' || FF_NAME)
RETURN MEASURE_UNDER, MEASURE_OVER
/* End Formula Text */
