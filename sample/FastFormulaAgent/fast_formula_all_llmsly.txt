 default for start_date_override(Date)  is   '01-JAN-1900'(DATE)
 default for end_date_override(Date)   is   '01-JAN-1900'(DATE)

  DEFAULT_DATA_VALUE FOR HWM_PUB_SCHD_MEASURE             IS 0
  DEFAULT_DATA_VALUE FOR HWM_PUB_SCHD_START_TIME          IS   '01-JAN-1900'(DATE)
  DEFAULT_DATA_VALUE FOR HWM_PUB_SCHD_STOP_TIME           IS   '01-JAN-1900'(DATE)

  DEFAULT_DATA_VALUE FOR HWM_EMP_SCHD_MEASURE             IS 0
  DEFAULT_DATA_VALUE FOR HWM_EMP_SCHD_START_DATE_TIME     IS  '01-JAN-1900'(DATE)
  DEFAULT_DATA_VALUE FOR HWM_EMP_SCHD_END_DATE_TIME       IS  '01-JAN-1900'(DATE)


INPUTS ARE
  start_date_override(DATE) ,
  end_date_override(DATE)

ffs_id = GET_CONTEXT(HWM_FFS_ID, 0)
rule_id = GET_CONTEXT(HWM_RULE_ID, 0)

ffName = 'WFM_UTIL_GET_EMPLOYMENT_SCHEDULE'
rLog  = add_rlog (ffs_id, rule_id, '>>> Enter: ' || ffName || ' - v105.1202 '  )

 rLog = add_rlog (ffs_id, rule_id, '(INPUTS : '  ||  ' , start_date_override='  ||  TO_CHAR( start_date_override ) ||
 ' , end_date_override='  ||  TO_CHAR( end_date_override ) ||   ' ) ' )


NullDate =  '01-JAN-1900'(DATE)
NullText =  '**FF_NULL**'


ctx_personId = GET_CONTEXT(HWM_RESOURCE_ID, 0)
ctx_subResource = GET_CONTEXT(HWM_SUBRESOURCE_ID, 0)
ctx_start_date = GET_CONTEXT(HWM_CTX_SEARCH_START_DATE,  NullDate)
ctx_end_date = GET_CONTEXT(HWM_CTX_SEARCH_END_DATE, NullDate)

 l_status = add_rlog (ffs_id , rule_id ,  '(UTIL_ ...context : '  ||
 ' , ctx_personId ='  ||  TO_CHAR( ctx_personId ) ||
 ' , ctx_subResource ='  ||  TO_CHAR( ctx_subResource ) ||
 ' , ctx_start_date ='  ||  TO_CHAR( ctx_start_date ) ||
 ' , ctx_end_date ='  ||  TO_CHAR( ctx_end_date ) ||
 ' ) ' )


stDate  = trunc(start_date_override)
endDate = trunc(end_date_override )
IF (start_date_override WAS DEFAULTED )   THEN (
	stDate = trunc(ctx_start_date)
)


IF (end_date_override WAS DEFAULTED )  THEN (
	endDate = trunc(ctx_end_date)
)
endDate = ADD_DAYS(trunc(endDate  ),1)


max_loop = 5000
pubCount = -1
empCount = -1

pubHrs = 0
EmpHrs = 0
TotalHrs = 0
HrsSource = NullText
nidx = 1

CHANGE_CONTEXTS(HWM_CTX_SEARCH_START_DATE = stDate,  HWM_CTX_SEARCH_END_DATE =  endDate)
(
	rLog  = add_rlog (ffs_id, rule_id, 'Final_Context:  subResource= '   || TO_CHAR( ctx_subResource ) ||
									'; personId ='  ||  TO_CHAR( ctx_personId ) ||
									'; stDate= '   || TO_CHAR( stDate ) ||
									'; endDate= '   || TO_CHAR( endDate ) )

		pubCount = HWM_PUB_SCHD_MEASURE.count
		rLog  = add_rlog (ffs_id, rule_id, '->Start_loop HWM_PUB_SCHD_MEASURE pubCount=' || TO_CHAR( pubCount )  )
		if (pubCount > 0 ) Then
		(
			nidx = 1
			HrsSource = 'PUB'
			 WHILE (HWM_PUB_SCHD_MEASURE.EXISTS(nidx))LOOP
			 (
				pubHrs = HWM_PUB_SCHD_MEASURE[nidx]
				TotalHrs = TotalHrs + pubHrs


				empPubDateStart = NullDate
				empPubDateEnd = NullDate
				if (HWM_PUB_SCHD_START_TIME.exists(nidx) ) then (
					empPubDateStart = HWM_PUB_SCHD_START_TIME[nidx]
				)

				if (HWM_PUB_SCHD_STOP_TIME.exists(nidx) ) then (
					empPubDateEnd = HWM_PUB_SCHD_STOP_TIME[nidx]
				)




					 rLog  = add_rlog (ffs_id, rule_id, 'HWM_PUB_SCHD__Row_number: '   || TO_CHAR( nidx ) ||
								  '; pubHrs= '   || TO_CHAR( pubHrs ) ||
								  '; TotalHrs= '   || TO_CHAR( TotalHrs ) ||
								  '; start_date_time= '   || TO_CHAR( empPubDateStart )  ||
								  '; end_date_time= '   || TO_CHAR( empPubDateEnd  )
							)

				nidx = nidx + 1
				if (nidx >  max_loop) Then (
					ex = raise_error (ffs_id, rule_id, 'WFM_UTIL_GET_EMPLOYMENT_SCHEDULE terminated due to possible end-less loop. ' ||
														'Number of rows for HWM_PUB_SCHD_MEASURE exceeds' ||  TO_CHAR(max_loop) )
				)

			)

		) else
		(
			empCount = HWM_EMP_SCHD_MEASURE.count
			rLog  = add_rlog (ffs_id, rule_id, '--------- start  HWM_EMP_SCHD_MEASURE loop no Change Contexts '   || 	';  empCount=' || TO_CHAR( empCount )  )
			nidx = 1
			HrsSource = 'EMP'
			 WHILE (HWM_EMP_SCHD_MEASURE.EXISTS(nidx))LOOP
			(
				EmpHrs = HWM_EMP_SCHD_MEASURE[nidx]
				TotalHrs = TotalHrs + EmpHrs

				empSchDateStart = NullDate
				empSchDateEnd = NullDate
				if (HWM_EMP_SCHD_START_DATE_TIME.exists(nidx) ) then (
					empSchDateStart = HWM_EMP_SCHD_START_DATE_TIME[nidx]
				)

				if (HWM_EMP_SCHD_END_DATE_TIME.exists(nidx) ) then (
					empSchDateEnd = HWM_EMP_SCHD_END_DATE_TIME[nidx]
				)

					 rLog  = add_rlog (ffs_id, rule_id, 'HWM_EMP_SCHD_MEASURE__Row_number: '   || TO_CHAR( nidx ) ||
								  '; EmpHrs= '   || TO_CHAR( EmpHrs ) ||
								  '; TotalHrs= '   || TO_CHAR( TotalHrs ) ||
								  '; start_date_time= '   || TO_CHAR( empSchDateStart  )  ||
								  '; end_date_time= '   || TO_CHAR( empSchDateEnd  )
							)


				nidx = nidx + 1
				if (nidx >  max_loop) Then (
				 ex = raise_error (ffs_id, rule_id, 'WFM_UTIL_GET_EMPLOYMENT_SCHEDULE terminated due to possible end-less loop. ' ||
													'Number of rows for HWM_EMP_SCHD_MEASURE exceeds' ||  TO_CHAR(max_loop) )
				)
			)

		)


)


rLog  = add_rlog (ffs_id, rule_id, '&lt;&lt;&lt; Exit:  ' || ffName ||
	' -  HrsSource=' || HrsSource || ' - TotalHrs=' || TO_CHAR( TotalHrs )  )


RETURN  TotalHrs, HrsSource
