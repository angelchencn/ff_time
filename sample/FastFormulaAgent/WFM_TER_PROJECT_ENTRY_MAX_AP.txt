 * Formula Name : WFM_TER_PROJECT_ENTRY_MAX_AP
 *
 * Formula Type:  Time Entry Rule (TER)
 *
 * Description:   Flags time entry details that exceed the defined limit for a
 *                specific project and returns an error message per offending entry.
 *
 * Detail:        Compares each reported time entry measure to the limit when
 *                the entry references the configured target project (by ID or
 *                number). Entries over the limit generate an error in the
 *                message array so the worker must correct the entry before
 *                submitting the time card.
 *
 *  Change History
 *  --------------
 *  Who               Ver      Date          Description
 *-----------------  ------   ------------  -------------------------------------
 * Codex Assistant     1.0    2026-03-23    Initial development based on user spec.
 *
 ***************************************************************************/

/*
 * Block 1: Default Declarations
 */
DEFAULT FOR HWM_CTXARY_RECORD_POSITIONS   IS  EMPTY_TEXT_NUMBER
DEFAULT FOR measure                       IS  EMPTY_NUMBER_NUMBER
DEFAULT FOR StartTime                     IS  EMPTY_DATE_NUMBER
DEFAULT FOR StopTime                      IS  EMPTY_DATE_NUMBER
DEFAULT FOR PayrollTimeType               IS  EMPTY_TEXT_NUMBER
DEFAULT_DATA_VALUE FOR PJC_PROJECT_ID     IS  0
DEFAULT_DATA_VALUE FOR PYR_Project        IS  '**FF_NULL**'

/*
 * Block 2: Inputs
 */
INPUTS ARE
  HWM_CTXARY_RECORD_POSITIONS,
  measure,
  StartTime,
  StopTime,
  PayrollTimeType,
  PJC_PROJECT_ID,
  PYR_Project

/*
 * Block 3: Context and Parameter Setup
 */
ffs_id = GET_CONTEXT(HWM_FFS_ID, 0)
rule_id = GET_CONTEXT(HWM_RULE_ID, 0)

ffName = 'WFM_TER_PROJECT_ENTRY_MAX_AP'
add_rlog (ffs_id, rule_id, 'Enter: ' || ffName)

NullDate = '01-JAN-1900'(DATE)
NullDateTime = '1900/01/01 00:00:00'(DATE)
NullText = '**FF_NULL**'
NullNumber = -99999
maxLoop = 1000

hSumLvl   = Get_Hdr_Text(rule_id, 'RUN_SUMMATION_LEVEL', 'DETAIL')
hExecType = Get_Hdr_Text(rule_id, 'RULE_EXEC_TYPE', 'CREATE')
processEmptyTc = Get_Hdr_Text(rule_id,'INCLUDE_EMPTY_TC', 'Y')
includeEmpty = 'N'
IF UPPER(processEmptyTc) = 'YES' OR UPPER(processEmptyTc) = 'Y' THEN(
  includeEmpty = 'Y'
)

pLimitHours      = GET_RVALUE_NUMBER(rule_id, 'DEFINED_LIMIT', 4)
pTargetProjectId = GET_RVALUE_NUMBER(rule_id, 'TARGET_PROJECT_ID', 0)
pTargetProjectNo = GET_RVALUE_TEXT(rule_id, 'TARGET_PROJECT_NUMBER', NullText)
pMessageCode     = GET_RVALUE_TEXT(rule_id, 'OVERRIDE_MESSAGE_CODE', 'HWM_FF_TER_PROJECT_DURATION_ERR')

add_rlog (
  ffs_id,
  rule_id,
  '(Parameters) Limit=' || TO_CHAR(pLimitHours) ||
  '; TargetProjectId=' || TO_CHAR(pTargetProjectId) ||
  '; TargetProjectNumber=' || pTargetProjectNo ||
  '; MessageCode=' || pMessageCode ||
  '; SumLevel=' || hSumLvl ||
  '; ExecType=' || hExecType
)

/*
 * Block 4: Processing Logic
 */
out_msg_ary = EMPTY_TEXT_NUMBER
maxAry = HWM_CTXARY_RECORD_POSITIONS.count
nidx = 0

WHILE (nidx < maxAry) LOOP
(
  nidx = nidx + 1
  entryMeasure = 0
  entryStartTime = NullDate
  entryProjectId = 0
  entryProjectNumber = NullText
  entryRecPos = NullText
  entryMsg = NullText
  durationHours = NullText
  hrs = 0
  mins = 0
  projectDisplay = NullText
  matchedProject = 'N'

  entryRecPos = HWM_CTXARY_RECORD_POSITIONS[nidx]
  IF (measure.exists(nidx)) THEN ( entryMeasure = measure[nidx] )
  IF (StartTime.exists(nidx)) THEN ( entryStartTime = StartTime[nidx] )
  IF (PJC_PROJECT_ID.exists(nidx)) THEN ( entryProjectId = PJC_PROJECT_ID[nidx] )
  IF (PYR_Project.exists(nidx)) THEN ( entryProjectNumber = PYR_Project[nidx] )

  IF (pTargetProjectId > 0 AND entryProjectId = pTargetProjectId) THEN (
    matchedProject = 'Y'
    projectDisplay = TO_CHAR(entryProjectId)
  )
  ELSE IF (pTargetProjectNo <> NullText AND entryProjectNumber <> NullText AND entryProjectNumber = pTargetProjectNo) THEN (
    matchedProject = 'Y'
    projectDisplay = entryProjectNumber
  )

  IF (matchedProject = 'Y' AND entryMeasure > pLimitHours) THEN (
    hrs = TRUNC(entryMeasure, 0)
    mins = TRUNC((entryMeasure - hrs) * 60, 0)
    durationHours = TO_CHAR(hrs) || 'h'
    IF (mins > 0) THEN (
      durationHours = durationHours || ' ' || TO_CHAR(mins) || 'm'
    )

    IF (projectDisplay = NullText) THEN (
      projectDisplay = TO_CHAR(entryProjectId)
    )

    l_tkn1 = 'PROJECT'
    l_val1 = projectDisplay
    l_tkn2 = 'ENTRY_DURATION'
    l_val2 = durationHours
    l_tkn3 = 'ENTRY_DATE'
    l_val3 = TO_CHAR(entryStartTime, 'YYYY-MM-DD')

    entryMsg = GET_OUTPUT_MSG3('HWM', pMessageCode, l_tkn1, l_val1, l_tkn2, l_val2, l_tkn3, l_val3)
    out_msg_ary[nidx] = entryMsg

    add_rlog (
      ffs_id,
      rule_id,
      'Entry ' || TO_CHAR(nidx) || ' exceeds limit. Project=' || projectDisplay ||
      '; Hours=' || TO_CHAR(entryMeasure) || '; Msg=' || entryMsg
    )
  )

  IF (nidx > maxLoop) THEN (
    raise_error(ffs_id, rule_id, 'Formula ' || ffName || ' terminated due to possible end-less loop.')
  )
)

add_rlog (ffs_id, rule_id, 'Exit: ' || ffName)

/*
 * Block 5: Outputs
 */
RETURN out_msg_ary
